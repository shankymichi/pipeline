
pipeline{
    agent any
    stages{
        
        
        stage('Build'){
            steps{
                
                    bat 'mvn clean install'
                
            }
        }
        stage('Test'){
            steps{
               
                    bat 'mvn test'
                
            }
        }
         stage('Package'){
            steps{
                
                    bat 'mvn package'
                
            }
        }
        stage('Deploy'){
            steps{
               
                    bat 'java -cp java-project-1.0-SNAPSHOT.jar com.sapient.hashing.App'
                
            }
        }
    }
}
